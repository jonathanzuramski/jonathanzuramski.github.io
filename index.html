    if(row[1]['play_type'] == 'pass' and row[1]['yards_gained'] == 0):
        zeropass += 1

team_data = pd.DataFrame(columns=['game_id','teamID', 'num_runs', 'run_yds', 'num_passes', 'pass_yds', 'win'])



l1 = data2019.groupby(['game_id', 'posteam', 'play_type'])['yards_gained'].sum().reset_index()
l1.rename(columns={'posteam': 'teamId'}, inplace=True)

l2 = data2019.groupby(['game_id', 'posteam'])['play_type'].value_counts().reset_index(name='number of plays')



passes = l2[l2['play_type'] == 'pass']['number of plays'].reset_index().drop(['index'],axis=1)
rushes = l2[l2['play_type'] == 'run']['number of plays'].reset_index().drop(['index'],axis=1)

l3 = l1.iloc[::2,:].reset_index().drop(['index'],axis=1)

team_data['teamID'] = l3['teamId']
team_data['game_id'] = l3['game_id']
team_data['pass_yds'] = ((l1[(l1['play_type']=='pass')])['yards_gained']).reset_index().drop(['index'],axis=1)
team_data['run_yds'] = ((l1[(l1['play_type']=='run')])['yards_gained']).reset_index().drop(['index'],axis=1)
team_data['num_runs'] = rushes
team_data['num_passes'] = passes
team_data['win'] = False

for row in win_data2019.iterrows():
    if(row[1]['home_score'] > row[1]['away_score']):
        team_data.loc[(team_data['game_id'] == row[1]['game_id']) & (team_data['teamID'] == row[1]['home_team']), 'win'] = True
    else:
        team_data.loc[(team_data['game_id'] == row[1]['game_id']) & (team_data['teamID'] == row[1]['away_team']), 'win'] = True

display(team_data)
